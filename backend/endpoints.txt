/api/users
GET /api/users/ # get user data from token
DELETE /api/users # delete user after password verification
POST /api/users/register # register new user
POST /api/users/password # password change
POST /api/users/login # login user
#endpoint for user data change


/api/subscriptions
GET /api/subscriptions # get data about subscription owned by user (from token)
POST /api/subscriptions/new # create new subscription for user
POST /api/subscriptions/renew # renew user subscription
POST /api/subscriptions/deactivate # deactivate user subscription
POST /api/subscriptions/join # user joining other user subscription
GET /api/subscriptions/share_code # create and get user subscription share code

/api/admins
POST /api/admins/login
POST /api/admins/register

/api/series
GET /api/series # get all series
GET /api/series/:genre # get all series form set genre
POST /api/series

GET /api/videos/ # get all videos
POST /api/videos/add_series # add multiple videos
POST /api/videos/find # search videos with multiple parameters

DELETE /api/videos/ # remove video
GET /api/videos/data/:filename # get single video data
GET /api/videos/:filename # get single video to watch


/api/watchlists
GET /api/watchlists # get user watchlist data
GET /api/watchlists/share # get user watchlist share code
POST /api/watchlists/shared # get another user watchlist data

/api/watchtogether
GET /api/watchtogether/share # get watchtogehther share code
POST /api/watchtogether/shared # join watchtogehter room
POSt /api/watchtogehter/video # chose video to watch
